# Chat P1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port the core chat interface from fs-whatsapp Vue to MagicFlow React — contacts, messaging, all message types, media, real-time WebSocket updates.

**Architecture:** React Query for all state (contacts, messages). WebSocket hook patches React Query cache for real-time updates. Plain scrollable div for messages (no virtualization). All components standalone and composable.

**Tech Stack:** React, React Query (`@tanstack/react-query`), WebSocket API, shadcn components

Create `hooks/use-websocket.ts`:

```typescript
"use client"

import { createContext, useContext, useEffect, useRef, useState, useCallback, type ReactNode } from "react"
import { getAccessToken } from "@/lib/auth"
import { useQueryClient } from "@tanstack/react-query"
import { contactKeys, messageKeys } from "@/hooks/queries/query-keys"

interface WebSocketContextValue {
  sendEvent: (type: string, payload?: any) => void
  subscribe: (eventType: string, handler: (payload: any) => void) => () => void
  isConnected: boolean
}

const WebSocketContext = createContext<WebSocketContextValue | null>(null)

export function useWebSocket(): WebSocketContextValue {
  const ctx = useContext(WebSocketContext)
  if (!ctx) throw new Error("useWebSocket must be used within WebSocketProvider")
  return ctx
}

const FAST_RETRY_MAX = 5
const FAST_RETRY_BASE_MS = 1000
const SLOW_RETRY_MS = 30000

export function WebSocketProvider({ children }: { children: ReactNode }) {
  const wsRef = useRef<WebSocket | null>(null)
  const listenersRef = useRef<Map<string, Set<(payload: any) => void>>>(new Map())
  const retryCountRef = useRef(0)
  const retryTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null)
  const [isConnected, setIsConnected] = useState(false)
  const isReconnectRef = useRef(false)
  const queryClient = useQueryClient()

  const connect = useCallback(() => {
    const token = getAccessToken()
    if (!token) return

    const wsUrl = `${process.env.NEXT_PUBLIC_FS_WHATSAPP_URL?.replace(/^http/, "ws")}/ws?token=${token}`
    const ws = new WebSocket(wsUrl)

    ws.onopen = () => {
      setIsConnected(true)
      retryCountRef.current = 0

      if (isReconnectRef.current) {
        // Refresh stale data after reconnection
        queryClient.invalidateQueries({ queryKey: contactKeys.all })
        queryClient.invalidateQueries({ queryKey: messageKeys.all })
      }
      isReconnectRef.current = true
    }

    ws.onmessage = (event) => {
      try {
        const msg = JSON.parse(event.data)
        if (msg.type === "ping") {
          ws.send(JSON.stringify({ type: "pong" }))
          return
        }
        // Dispatch to subscribers
        const handlers = listenersRef.current.get(msg.type)
        if (handlers) {
          handlers.forEach((handler) => handler(msg.payload))
        }
        // Also dispatch status_update for message_status events (backend uses both names)
        if (msg.type === "message_status") {
          const statusHandlers = listenersRef.current.get("status_update")
          if (statusHandlers) {
            statusHandlers.forEach((handler) => handler(msg.payload))
          }
        }
      } catch (e) {
        console.error("[WebSocket] Failed to parse message:", e)
      }
    }

    ws.onclose = () => {
      setIsConnected(false)
      wsRef.current = null
      scheduleReconnect()
    }

    ws.onerror = () => {
      ws.close()
    }

    wsRef.current = ws
  }, [queryClient])

  const scheduleReconnect = useCallback(() => {
    if (retryTimerRef.current) clearTimeout(retryTimerRef.current)

    const count = retryCountRef.current
    const delay = count < FAST_RETRY_MAX
      ? FAST_RETRY_BASE_MS * Math.pow(2, count)
      : SLOW_RETRY_MS

    retryTimerRef.current = setTimeout(() => {
      retryCountRef.current++
      connect()
    }, delay)
  }, [connect])

  const sendEvent = useCallback((type: string, payload?: any) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({ type, payload }))
    }
  }, [])

  const subscribe = useCallback((eventType: string, handler: (payload: any) => void) => {
    if (!listenersRef.current.has(eventType)) {
      listenersRef.current.set(eventType, new Set())
    }
    listenersRef.current.get(eventType)!.add(handler)

    return () => {
      listenersRef.current.get(eventType)?.delete(handler)
    }
  }, [])

  useEffect(() => {
    connect()
    return () => {
      if (retryTimerRef.current) clearTimeout(retryTimerRef.current)
      wsRef.current?.close()
    }
  }, [connect])

  return (
    <WebSocketContext.Provider value={{ sendEvent, subscribe, isConnected }}>
      {children}
    </WebSocketContext.Provider>
  )
}
```

```typescript
import { useInfiniteQuery, useQuery } from "@tanstack/react-query"
import { apiClient } from "@/lib/api-client"
import { contactKeys } from "./query-keys"
import type { Contact, ContactsResponse } from "@/types/chat"

const PAGE_SIZE = 20

interface UseContactsOptions {
  search?: string
  channel?: "whatsapp" | "instagram" | null
}

export function useContacts(options: UseContactsOptions = {}) {
  const { search, channel } = options

  return useInfiniteQuery({
    queryKey: contactKeys.list({ search, channel: channel ?? undefined }),
    queryFn: async ({ pageParam = 1 }) => {
      const params = new URLSearchParams()
      params.set("page", String(pageParam))
      params.set("limit", String(PAGE_SIZE))
      if (search) params.set("search", search)
      if (channel) params.set("channel", channel)

      return apiClient.get<ContactsResponse>(`/api/contacts?${params}`)
    },
    getNextPageParam: (lastPage) =>
      lastPage.contacts.length === PAGE_SIZE ? lastPage.page + 1 : undefined,
    initialPageParam: 1,
  })
}

export function useContact(id: string | null) {
  return useQuery({
    queryKey: contactKeys.detail(id!),
    queryFn: () => apiClient.get<Contact>(`/api/contacts/${id}`),
    enabled: !!id,
  })
}
```

```typescript
import { useInfiniteQuery, useMutation, useQueryClient } from "@tanstack/react-query"
import { apiClient } from "@/lib/api-client"
import { messageKeys, contactKeys } from "./query-keys"
import type { Message, MessagesResponse } from "@/types/chat"

const PAGE_SIZE = 30

export function useMessages(contactId: string | null) {
  return useInfiniteQuery({
    queryKey: messageKeys.list(contactId!),
    queryFn: async ({ pageParam }) => {
      const params = new URLSearchParams()
      params.set("limit", String(PAGE_SIZE))
      if (pageParam) params.set("before_id", pageParam)

      return apiClient.get<MessagesResponse>(
        `/api/contacts/${contactId}/messages?${params}`
      )
    },
    getNextPageParam: (lastPage) => {
      if (!lastPage.has_more || lastPage.messages.length === 0) return undefined
      return lastPage.messages[lastPage.messages.length - 1].id
    },
    initialPageParam: undefined as string | undefined,
    enabled: !!contactId,
    staleTime: Infinity, // CRITICAL: fetching marks messages as read — prevent background refetches
  })
}

export function useSendMessage(contactId: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (text: string) => {
      return apiClient.post<Message>(`/api/contacts/${contactId}/messages`, {
        type: "text",
        content: { body: text },
      })
    },
    onSuccess: (newMessage) => {
      // Append to first page of messages cache
      queryClient.setQueryData(
        messageKeys.list(contactId),
        (old: any) => {
          if (!old) return old
          const firstPage = old.pages[0]
          // Dedup: check if message already exists (from WebSocket)
          const exists = firstPage.messages.some((m: Message) => m.id === newMessage.id)
          if (exists) return old
          return {
            ...old,
            pages: [
              { ...firstPage, messages: [newMessage, ...firstPage.messages] },
              ...old.pages.slice(1),
            ],
          }
        }
      )
      // Update contact's last message in contacts cache
      queryClient.invalidateQueries({ queryKey: contactKeys.lists() })
    },
  })
}

export function useSendMedia(contactId: string) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async ({ file, type, caption }: { file: File; type: string; caption?: string }) => {
      const formData = new FormData()
      formData.append("file", file)
      formData.append("contact_id", contactId)
      formData.append("type", type)
      if (caption) formData.append("caption", caption)

      return apiClient.post<Message>("/api/messages/media", formData)
    },
    onSuccess: (newMessage) => {
      queryClient.setQueryData(
        messageKeys.list(contactId),
        (old: any) => {
          if (!old) return old
          const firstPage = old.pages[0]
          const exists = firstPage.messages.some((m: Message) => m.id === newMessage.id)
          if (exists) return old
          return {
            ...old,
            pages: [
              { ...firstPage, messages: [newMessage, ...firstPage.messages] },
              ...old.pages.slice(1),
            ],
          }
        }
      )
      queryClient.invalidateQueries({ queryKey: contactKeys.lists() })
    },
  })
}
```

```typescript
"use client"

import { useState, useCallback } from "react"
import { useSearchParams, useRouter } from "next/navigation"

export function useChat() {
  const searchParams = useSearchParams()
  const router = useRouter()
  const [isAtBottom, setIsAtBottom] = useState(true)

  const activeContactId = searchParams.get("contact")

  const setActiveContact = useCallback((contactId: string | null) => {
    const params = new URLSearchParams(searchParams.toString())
    if (contactId) {
      params.set("contact", contactId)
    } else {
      params.delete("contact")
    }
    router.push(`/chat?${params}`)
  }, [searchParams, router])

  return {
    activeContactId,
    setActiveContact,
    isAtBottom,
    setIsAtBottom,
  }
}
```

```typescript
function formatRelativeTime(dateString: string): string {
  const date = new Date(dateString)
  const now = new Date()
  const diffMs = now.getTime() - date.getTime()
  const diffMins = Math.floor(diffMs / 60000)
  const diffHours = Math.floor(diffMs / 3600000)
  const diffDays = Math.floor(diffMs / 86400000)

  if (diffMins < 1) return "now"
  if (diffMins < 60) return `${diffMins}m`
  if (diffHours < 24) return `${diffHours}h`
  if (diffDays < 7) return `${diffDays}d`
  return date.toLocaleDateString(undefined, { month: "short", day: "numeric" })
}
```

```typescript
"use client"

import { useRef, useCallback, useEffect } from "react"
import { useMessages } from "@/hooks/queries/use-messages"
import { MessageBubble } from "./message-bubble"
import type { Message } from "@/types/chat"

interface MessageListProps {
  contactId: string
  isAtBottom: boolean
  onAtBottomChange: (atBottom: boolean) => void
}

export function MessageList({ contactId, isAtBottom, onAtBottomChange }: MessageListProps) {
  const scrollRef = useRef<HTMLDivElement>(null)
  const { data, isLoading, hasNextPage, fetchNextPage, isFetchingNextPage } = useMessages(contactId)

  // Flatten pages — newest messages first in page 0
  const messages: Message[] = data?.pages.flatMap((p) => p.messages) ?? []
  // Reverse for display: oldest at top, newest at bottom
  const displayMessages = [...messages].reverse()

  // Scroll to bottom on initial load and new messages (when at bottom)
  useEffect(() => {
    if (isAtBottom && scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight
    }
  }, [messages.length, isAtBottom])

  // Infinite scroll upward — load older messages
  const handleScroll = useCallback(async () => {
    const el = scrollRef.current
    if (!el) return

    // Track isAtBottom
    const atBottom = el.scrollTop + el.clientHeight >= el.scrollHeight - 50
    onAtBottomChange(atBottom)

    // Load older when near top
    if (el.scrollTop < 100 && hasNextPage && !isFetchingNextPage) {
      const prevScrollHeight = el.scrollHeight
      const prevScrollTop = el.scrollTop
      await fetchNextPage()
      // Preserve scroll position after prepend
      requestAnimationFrame(() => {
        if (el) {
          el.scrollTop = el.scrollHeight - prevScrollHeight + prevScrollTop
        }
      })
    }
  }, [hasNextPage, isFetchingNextPage, fetchNextPage, onAtBottomChange])

  // Date separator helper
  const getDateLabel = (dateString: string) => {
    const date = new Date(dateString)
    const today = new Date()
    const yesterday = new Date(today)
    yesterday.setDate(yesterday.getDate() - 1)

    if (date.toDateString() === today.toDateString()) return "Today"
    if (date.toDateString() === yesterday.toDateString()) return "Yesterday"
    return date.toLocaleDateString(undefined, { month: "long", day: "numeric", year: "numeric" })
  }

  if (isLoading) {
    return (
      <div className="flex-1 flex items-center justify-center">
        <div className="animate-spin h-6 w-6 border-2 border-primary border-t-transparent rounded-full" />
      </div>
    )
  }

  return (
    <div ref={scrollRef} onScroll={handleScroll} className="flex-1 overflow-y-auto px-4 py-2">
      {isFetchingNextPage && (
        <div className="flex justify-center py-2">
          <div className="animate-spin h-4 w-4 border-2 border-primary border-t-transparent rounded-full" />
        </div>
      )}

      {displayMessages.map((message, index) => {
        const prevMessage = index > 0 ? displayMessages[index - 1] : null
        const showDate = !prevMessage ||
          new Date(message.created_at).toDateString() !== new Date(prevMessage.created_at).toDateString()

        return (
          <div key={message.id}>
            {showDate && (
              <div className="flex justify-center my-3">
                <span className="text-xs text-muted-foreground bg-muted px-3 py-1 rounded-full">
                  {getDateLabel(message.created_at)}
                </span>
              </div>
            )}
            <MessageBubble message={message} />
          </div>
        )
      })}
    </div>
  )
}
```

```typescript
import { useState, useRef, useCallback } from "react"
import { Send, Paperclip } from "lucide-react"

interface MessageInputProps {

}

export function MessageInput({ contactId, onAttachClick }: MessageInputProps) {
  const [text, setText] = useState("")
  const textareaRef = useRef<HTMLTextAreaElement>(null)
  const { mutate: sendMessage, isPending } = useSendMessage(contactId)

  const handleSend = useCallback(() => {
    const trimmed = text.trim()
    if (!trimmed || isPending) return
    sendMessage(trimmed, {
      onSuccess: () => {
        setText("")
        if (textareaRef.current) {
          textareaRef.current.style.height = "auto"
        }
      },
    })
  }, [text, isPending, sendMessage])

  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault()
      handleSend()
    }
  }, [handleSend])

  const handleInput = useCallback(() => {
    const textarea = textareaRef.current
    if (textarea) {
      textarea.style.height = "auto"
      textarea.style.height = Math.min(textarea.scrollHeight, 120) + "px"
    }
  }, [])

  return (
    <div className="flex items-end gap-2 px-4 py-3 border-t">
      <Button variant="ghost" size="icon" onClick={onAttachClick} className="flex-shrink-0 cursor-pointer">
        <Paperclip className="h-4 w-4" />
      </Button>

      <textarea
        ref={textareaRef}
        value={text}
        onChange={(e) => { setText(e.target.value); handleInput() }}
        onKeyDown={handleKeyDown}
        placeholder="Type a message..."
        rows={1}
        className="flex-1 resize-none rounded-lg border bg-background px-3 py-2 text-sm focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-ring"
      />

          <div className="animate-spin h-4 w-4 border-2 border-primary-foreground border-t-transparent rounded-full" />
        ) : (
          <Send className="h-4 w-4" />
```

- [ ] **Step 1: Create chat-specific WebSocket event handler**

Create `hooks/use-chat-websocket.ts`:

```typescript
"use client"

import { useEffect, useCallback } from "react"
import { useQueryClient } from "@tanstack/react-query"
import { useWebSocket } from "./use-websocket"
import { contactKeys, messageKeys } from "./queries/query-keys"
import type { Message, Contact } from "@/types/chat"

export function useChatWebSocket(activeContactId: string | null) {
  const { subscribe, sendEvent } = useWebSocket()
  const queryClient = useQueryClient()

  // Handle new_message events
  useEffect(() => {
    const unsub = subscribe("new_message", (payload: any) => {
      const message = payload as Message
      const isActiveContact = message.contact_id === activeContactId

      // Update message cache for active contact (with dedup)
      if (isActiveContact) {
        queryClient.setQueryData(
          messageKeys.list(message.contact_id),
          (old: any) => {
            if (!old) return old
            const firstPage = old.pages[0]
            if (!firstPage) return old

            // Dedup
            const exists = old.pages.some((p: any) =>
              p.messages.some((m: Message) => m.id === message.id)
            )
            if (exists) return old

            // Insert by timestamp (newest first in page 0)
            const updatedMessages = [message, ...firstPage.messages]
              .sort((a, b) => new Date(b.created_at).getTime() - new Date(a.created_at).getTime())

            return {
              ...old,
              pages: [
                { ...firstPage, messages: updatedMessages },
                ...old.pages.slice(1),
              ],
            }
          }
        )
      }

      // Update contact list cache (reorder to top, update preview/unread)
      queryClient.setQueryData(
        contactKeys.lists(),
        (old: any) => {
          if (!old) return old

          // Flatten all pages
          const allContacts: Contact[] = old.pages.flatMap((p: any) => p.contacts)

          // Find and update the contact
          const idx = allContacts.findIndex((c) => c.id === message.contact_id)
          let contact: Contact
          if (idx !== -1) {
            contact = {
              ...allContacts[idx],
              last_message_preview: message.content?.body || "",
              last_message_at: message.created_at,
              unread_count: isActiveContact
                ? allContacts[idx].unread_count
                : allContacts[idx].unread_count + 1,
            }
            allContacts.splice(idx, 1)
          } else {
            // Contact not in cache — will appear on next refetch
            return old
          }

          // Prepend to top
          allContacts.unshift(contact)

          // Merge into page 0, keep last page intact
          const lastPage = old.pages[old.pages.length - 1]
          return {
            ...old,
            pages: old.pages.length === 1
              ? [{ ...old.pages[0], contacts: allContacts }]
              : [
                  { ...old.pages[0], contacts: allContacts.slice(0, -lastPage.contacts.length) },
                  lastPage,
                ],
            }
        }
      )
    })

    return unsub
  }, [subscribe, activeContactId, queryClient])

  // Handle status_update events (covers both "status_update" and "message_status" via WebSocket provider normalization)
  useEffect(() => {
    const unsub = subscribe("status_update", (payload: any) => {
      const { message_id, status, contact_id } = payload
      const targetContactId = contact_id || activeContactId
      if (!targetContactId) return

      queryClient.setQueryData(
        messageKeys.list(targetContactId),
        (old: any) => {
          if (!old) return old
          return {
            ...old,
            pages: old.pages.map((page: any) => ({
              ...page,
              messages: page.messages.map((m: Message) =>
                m.id === message_id ? { ...m, status } : m
              ),
            })),
          }
        }
      )
    })

    return unsub
  }, [subscribe, activeContactId, queryClient])
}
```
